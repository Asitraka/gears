193 {
194  // process default text geometry tags
195  G4bool processed = G4tgrLineProcessor::ProcessLine(words);
196  if (processed) return true;
197 
198  // process added tags
199  G4String tag = words[0];
200  tag.toLower();
201  if (tag.substr(0,5)==":prop") {
202  G4NistManager* mgr = G4NistManager::Instance();
203  G4cout<<"Set optical properties of "<<words[1]<<":"<<G4endl;
204  mgr->FindOrBuildMaterial(words[1])
205  ->SetMaterialPropertiesTable(CreateMaterialPropertiesTable(words,2));
206  return true;
207  } else if (tag.substr(0,5)==":surf") {
208  BorderSurface *bdr = new BorderSurface;
209  bdr->next=Border;
210  Border=bdr;
211  bdr->name=words[1];
212  bdr->v1=words[2];
213  bdr->v2=words[3];
214  bdr->optic = new G4OpticalSurface(words[1]);
215  size_t i=4;
216  // loop over optical surface setup lines
217  while (i<words.size()) {
218  G4String setting = words[i], value = words[i+1];
219  setting.toLower(); value.toLower();
220  if (setting=="property") {
221  i++;
222  break;
223  } else if(setting=="type") {
224  if (value=="dielectric_metal")
225  bdr->optic->SetType(dielectric_metal);
226  else if(value=="dielectric_dielectric")
227  bdr->optic->SetType(dielectric_dielectric);
228  else if(value=="firsov") bdr->optic->SetType(firsov);
229  else if(value=="x_ray") bdr->optic->SetType(x_ray);
230  else G4cout<<"Unknown surface type "<<value<<", ignored!"<<G4endl;
231  } else if(setting=="model") {
232  if (value=="glisur") bdr->optic->SetModel(glisur);
233  else if(value=="unified") bdr->optic->SetModel(unified);
234  else G4cout<<"Unknown surface model "<<value<<", ignored!"<<G4endl;
235  } else if(setting=="finish") {
236  if(value=="polished") bdr->optic->SetFinish(polished);
237  else if(value=="polishedfrontpainted")
238  bdr->optic->SetFinish(polishedfrontpainted);
239  else if(value=="polishedbackpainted")
240  bdr->optic->SetFinish(polishedbackpainted);
241  else if(value=="ground") bdr->optic->SetFinish(ground);
242  else if(value=="groundfrontpainted")
243  bdr->optic->SetFinish(groundfrontpainted);
244  else if(value=="groundbackpainted")
245  bdr->optic->SetFinish(groundbackpainted);
246  else

248  } else if(setting=="sigma_alpha") {

253  }

255  G4cout<<"Set optical properties of "<<bdr->name<<":"<<G4endl;
