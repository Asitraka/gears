325 #include <G4NistManager.hh>
326 #include <G4tgbMaterialMgr.hh>
327 G4bool LineProcessor::ProcessLine(const std::vector<G4String> &words)
328 {
329  // process default text geometry tags
330  G4bool processed = G4tgrLineProcessor::ProcessLine(words);
331  if (processed) return true; // no new tags involved
332 
333  // process added tags: prop & surf
334  G4String tag = words[0];
335  tag.toLower(); // to lower cases
336  if (tag.substr(0,5)==":prop") { // set optical properties of a material
337  G4NistManager *mgr = G4NistManager::Instance(); mgr->SetVerbose(2);
338  G4Material *m = mgr->FindOrBuildMaterial(words[1]);
339  if (m==NULL) // if not in NIST, then build in tgb
340  m=G4tgbMaterialMgr::GetInstance()->FindOrBuildG4Material(words[1]);
341  G4cout<<"GEARS: Set optical properties of "<<words[1]<<":"<<G4endl;
342  m->SetMaterialPropertiesTable(CreateMaterialPropertiesTable(words,2));
343  return true;
344  } else if (tag.substr(0,5)==":surf") { // define an optical surface
345  BorderSurface *bdr = new BorderSurface;
346  bdr->next=Border; // save current border pointer
347  Border=bdr; // overwrite current border pointer
348  bdr->name=words[1];
349  bdr->v1=words[2];
350  bdr->v2=words[3];
351  bdr->optic = new G4OpticalSurface(words[1]);
352  size_t i=4;
353  // loop over optical surface setup lines
354  while (i<words.size()) {
355  G4String setting = words[i], value = words[i+1];
356  setting.toLower(); value.toLower();
357  if (setting=="property") {
358  i++;
359  break;
360  } else if(setting=="type") {
361  if (value=="dielectric_metal")
362  bdr->optic->SetType(dielectric_metal);
363  else if(value=="dielectric_dielectric")
364  bdr->optic->SetType(dielectric_dielectric);
365  else if(value=="firsov") bdr->optic->SetType(firsov);
366  else if(value=="x_ray") bdr->optic->SetType(x_ray);
367  else G4cout<<"GERAS: Ignore unknown surface type "<<value<<G4endl;
368  } else if(setting=="model") {
369  if (value=="glisur") bdr->optic->SetModel(glisur);
370  else if(value=="unified") bdr->optic->SetModel(unified);
371  else G4cout<<"GERAS: Ignore unknown surface model "<<value<<G4endl;
372  } else if(setting=="finish") {
373  if(value=="polished") bdr->optic->SetFinish(polished);
374  else if(value=="polishedfrontpainted")
375  bdr->optic->SetFinish(polishedfrontpainted);
376  else if(value=="polishedbackpainted")
377  bdr->optic->SetFinish(polishedbackpainted);
378  else if(value=="ground") bdr->optic->SetFinish(ground);
379  else if(value=="groundfrontpainted")
380  bdr->optic->SetFinish(groundfrontpainted);
381  else if(value=="groundbackpainted")
382  bdr->optic->SetFinish(groundbackpainted);
383  else
384  G4cout<<"GERAS: Ignore unknown surface finish "<<value<<G4endl;
385  } else if(setting=="sigma_alpha") {
386  bdr->optic->SetSigmaAlpha(G4UIcommand::ConvertToInt(value));
387  } else
388  G4cout<<"GERAS: Ignore unknown surface setting "<<value<<G4endl;
389  i+=2;
390  }
391  if (i<words.size()) { // break while loop because of "property"
392  G4cout<<"GEARS: Set optical properties of "<<bdr->name<<":"<<G4endl;
393  bdr->optic->SetMaterialPropertiesTable(
394  CreateMaterialPropertiesTable(words,i));
395  }
396  return true;
397  } else
398  return false;
399 }

402 #include <G4tgrUtils.hh>
403 G4MaterialPropertiesTable* LineProcessor::CreateMaterialPropertiesTable(
404  const std::vector<G4String> &words, size_t idxOfWords)
405 {
406  bool photonEnergyUnDefined=true;
407  int cnt=0; // number of photon energy values
408  double *energies; // photon energy values
409  G4MaterialPropertiesTable *table = new G4MaterialPropertiesTable();
410  for (size_t i=idxOfWords; i<words.size(); i++) {
411  G4String property = words[i]; property.toUpper();
412  if (property=="SCINTILLATIONYIELD" || property=="RESOLUTIONSCALE"
413  || property=="FASTTIMECONSTANT" || property=="SLOWTIMECONSTANT"
414  || property=="YIELDRATIO" || property=="WLSTIMECONSTANT") {
415  table->AddConstProperty(property, G4tgrUtils::GetDouble(words[i+1]));
416  G4cout<<"GEARS: "<<property<<"="<<words[i+1]<<G4endl;
417  i++; // property value has been used
418  } else if (property.substr(0,12)=="PHOTON_ENERG") {
419  photonEnergyUnDefined=false;
420  cnt = G4UIcommand::ConvertToInt(words[i+1]); // get array size
421  energies = new double[cnt]; // create energy array
422  for (int j=0; j<cnt; j++)
423  energies[j]=G4tgrUtils::GetDouble(words[i+2+j]);
424  i=i+1+cnt; // array has been used
425  } else { // wavelength-dependent properties
426  if (photonEnergyUnDefined) {
427  G4cout<<"GEARS: photon energies undefined, "
428  <<"ignore all wavelength-dependent properties!"<<G4endl;
429  break;
430  }
431  double *values = new double[cnt];
432  for (int j=0; j<cnt; j++)
433  values[j]=G4tgrUtils::GetDouble(words[i+1+j]);
434  G4cout<<"GEARS: "<<property<<"="<<values[0]<<", "
435  <<values[1]<<"..."<<G4endl;
436  table->AddProperty(property, energies, values, cnt);
437  delete[] values;
438  i=i+cnt;
439  }
440  }
441  delete[] energies;
442  return table;
443 }
